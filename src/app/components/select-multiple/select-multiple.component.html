Options select All Options Function for Select Multiple Nodos

            
    allOptionsSelected: boolean = false;

   checkedNodesList(event: IListLCTSelectMultiple[]) {
    const selected = event.find((e) => e.name);
    const optionsSelected = this.options.find((node: any) => node.value === 'all')!;
    if (optionsSelected?.checked && !this.allOptionsSelected && selected?.value === 'all') {
      this.allOptionsSelected = true;
      this.options.forEach((node: any) => {
        node.checked = true;
      });
    }
    if (!optionsSelected?.checked && this.allOptionsSelected) {
      this.allOptionsSelected = false;
      this.options.forEach((node: any) => {
        node.checked = false;
      });
    }
    if (
      this.allOptionsSelected &&
      this.options
        .filter((node: any) => node.value !== 'all')
        .some((node: any) => !node.checked)
    ) {
      this.allOptionsSelected = false;
      optionsSelected.checked = false;
    }
    if (
      !this.allOptionsSelected &&
      this.options
        .filter((node: any) => node.value !== 'all')
        .every((node: any) => node.checked)
    ) {
      this.allOptionsSelected = true;
      optionsSelected.checked = true;
    }
  }
